  3 // Polyfill for requestAnimationFrame() / cancelAnimationFrame()
  4 (function() {
  5   var lastTime = 0;
  6   var vendors = ['webkit', 'moz', 'ms', 'o'];
  7   
  8   for (var i = 0; i < vendors.length && !window.requestAnimationFrame; i++) {
  9     window.requestAnimationFrame = window[vendors[i] + 'RequestAnimationFrame'];
 10     window.cancelAnimationFrame = window[vendors[i] + 'CancelAnimationFrame'] ||
 11       window[vendors[i] + 'RequestCancelAnimationFrame'];
 12   }
 13 
 14   if (!window.requestAnimationFrame) {
 15     window.requestAnimationFrame = function(callback, element) {
 16       var currTime = (Date['now']) ? Date.now() : +new Date();
 17       var timeToCall = Math.max(0, 16 - (currTime - lastTime));
 18       var id = window.setTimeout(function() {
 19         callback(currTime + timeToCall);
 20       }, timeToCall);
 21       lastTime = currTime + timeToCall;
 22       return id;
 23     };
 24   }
 25 
 26   if (!window.cancelAnimationFrame) {
 27     window.cancelAnimationFrame = function(id) {
 28       window.clearTimeout(id);
 29     };
 30   }
 31 })();

 47 var SKScrollView = function(element) {
 48   var kBounceTransitionDuration = 0.35;
 49   var kAccelerationTimeout = 250;
 50   var kAcceleration = 15;
 51   var kElasticDeceleration = 0.03;
 52   var kElasticAcceleration = 0.08;
 53   var kMinimumDecelerationVelocity = 1;
 54   var kDecelerationFactor = 0.95;
 55   var kMinimumVelocity = 0.01;
 56   var kMinimumPageTurnVelocity = 5;
 57   var kScrollBarHideTimeout = 100;
 58   
 59   var _isDragging = false;
 60   var _isDecelerating = false;
 61   var _startAccelerateX = 0;
 62   var _startAccelerateY = 0;
 63   var _lastMouseX = -1;
 64   var _lastMouseY = -1;
 65   var _lastTouchIdentifier = 0;
 66   var _lastTimeStamp = Date.now();
 67   var _decelerationAnimationInterval = null;
 68   var _mouseWheelTimeout = null;
 69   
 70   var $window = $(window['addEventListener'] ? window : document.body);
 71   
 72   var $element = this.$element = $(element);
 73   element = this.element = $(element).get(0);
 74   
 75   var scrollView = element.scrollView;
 76   if (scrollView) return scrollView;
 77   
 78   element.scrollView = this;
 79   
 80   var self = this;
 81   var content = this.content = new SKScrollContent(this);
 82   var horizontalScrollBar = this.horizontalScrollBar = new SKScrollBar(this, SKScrollBarType.Horizontal);
 83   var verticalScrollBar = this.verticalScrollBar = new SKScrollBar(this, SKScrollBarType.Vertical);
 84   
 85   var bounceTransitionDuration = this.bounceTransitionDuration = kBounceTransitionDuration;
 86   
 87   var alwaysBounceHorizontal = $element.attr('data-always-bounce-horizontal') || 'false';
 88   var alwaysBounceVertical = $element.attr('data-always-bounce-vertical') || 'false';
 89   alwaysBounceHorizontal = this.alwaysBounceHorizontal = this.alwaysBounceHorizontal || (alwaysBounceHorizontal !== 'false');
 90   alwaysBounceVertical = this.alwaysBounceVertical = this.alwaysBounceVertical || (alwaysBounceVertical !== 'false');
 91   
 92   var showsHorizontalScrollIndicator = $element.attr('data-shows-horizontal-scroll-indicator') || 'true';
 93   var showsVerticalScrollIndicator = $element.attr('data-shows-vertical-scroll-indicator') || 'true';
 94   showsHorizontalScrollIndicator = this.showsHorizontalScrollIndicator = this.showsHorizontalScrollIndicator && (showsHorizontalScrollIndicator !== 'false');
 95   showsVerticalScrollIndicator = this.showsVerticalScrollIndicator = this.showsVerticalScrollIndicator && (showsVerticalScrollIndicator !== 'false');
 96   
 97   var pagingEnabled = $element.attr('data-paging-enabled') || 'false';
 98   pagingEnabled = this.pagingEnabled = this.pagingEnabled || (pagingEnabled !== 'false');
 99   
100   var startDeceleration = function(startTime) {
101     var acceleration = (startTime - _lastTimeStamp) / kAcceleration;
102     var accelerateDeltaX = self.x - _startAccelerateX;
103     var accelerateDeltaY = self.y - _startAccelerateY;
104     var velocityX = accelerateDeltaX / acceleration;
105     var velocityY = accelerateDeltaY / acceleration;
106     var minimumX = self.minimumX;
107     var minimumY = self.minimumY;
108     var elasticDeceleration = kElasticDeceleration;
109     var elasticAcceleration = kElasticAcceleration;
110     var lastFrameTime = 0;
111     
112     var stepAnimation = function (currentFrameTime) {
113       if (!_isDecelerating) return;
114       
115       var deltaTime = currentFrameTime - lastFrameTime;
116       var x = self.x = self.x + (!self.canScrollHorizontal() ? 0 : velocityX);
117       var y = self.y = self.y + (!self.canScrollVertical() ? 0 : velocityY);
118       
119       velocityX *= kDecelerationFactor;
120       velocityY *= kDecelerationFactor;
121       
122       if (Math.abs(velocityX) <= kMinimumVelocity &&
123           Math.abs(velocityY) <= kMinimumVelocity) {
124         _isDecelerating = false;
125         scrollEnd();
126         return;
127       }
128       
129       content.translate(x, y);
130       _decelerationAnimationInterval = window.requestAnimationFrame(stepAnimation);
131       
132       var elasticX = (x < minimumX) ? minimumX - x : (x > 0) ? -x : 0;
133       var elasticY = (y < minimumY) ? minimumY - y : (y > 0) ? -y : 0;
134       
135       if (elasticX) velocityX = (elasticX * velocityX <= 0) ?
136         velocityX + (elasticX * elasticDeceleration) : elasticX * elasticAcceleration;
137       
138       if (elasticY) velocityY = (elasticY * velocityY <= 0) ?
139         velocityY + (elasticY * elasticDeceleration) : elasticY * elasticAcceleration;
140       
141       lastFrameTime = currentFrameTime;
142     };
143     
144     if (Math.abs(velocityX) > kMinimumDecelerationVelocity ||
145         Math.abs(velocityY) > kMinimumDecelerationVelocity) {
146       _isDecelerating = true;
147       _decelerationAnimationInterval = window.requestAnimationFrame(stepAnimation);
148       lastFrameTime = Date.now();
149     } else {
150       scrollEnd();
151     }
152   };
153   
154   var stopDeceleration = function() {
155     _isDecelerating = false;
156     window.cancelAnimationFrame(_decelerationAnimationInterval);
157   };
158   
159   var scrollStart = function() {
160     if (self.isScrolling) return;
161     self.isScrolling = true;
162     
163     if (self.canScrollHorizontal()) horizontalScrollBar.show();
164     if (self.canScrollVertical()) verticalScrollBar.show();
165     
166     $element.trigger(SKScrollEventType.ScrollStart);
167     $('input:focus').blur();
168   };
169   
170   var scrollEnd = function() {
171     if (!self.isScrolling) return;
172     self.isScrolling = false;
173     
174     content.translate(self.x = Math.round(self.x), self.y = Math.round(self.y));
175     
176     if (self.canScrollHorizontal()) horizontalScrollBar.hide();
177     if (self.canScrollVertical()) verticalScrollBar.hide();
178     
179     $element.trigger(SKScrollEventType.ScrollEnd);
180   };
181   
182   var bounceTransitionEndHandler = function(evt) {
183     $element.unbind(evt);
184     scrollEnd();
185   };
186   
187   $element.bind('mousewheel DOMMouseScroll', function(evt) {
188     window.clearTimeout(_mouseWheelTimeout);
189     
190     var originalEvent = evt.originalEvent;
191     var detail = originalEvent.detail;
192     var delta = originalEvent.wheelDelta;
193     var normalizedDelta = (detail) ? ((delta) ? ((delta / detail / 40 * detail > 0) ? 1 : -1) : -detail / 3) : delta / 120;
194     
195     var scrollViewSize = self.getSize();
196     var contentSize = content.getSize();
197     
198     self.minimumX = scrollViewSize.width - contentSize.width;
199     self.minimumY = scrollViewSize.height - contentSize.height;
200     
201     var deltaX = 0;
202     var deltaY = normalizedDelta * 8;
203     
204     var x = self.x + deltaX;
205     var y = self.y + deltaY;
206     
207     x = (x < self.minimumX) ? self.minimumX : (x > 0) ? 0 : x;
208     y = (y < self.minimumY) ? self.minimumY : (y > 0) ? 0 : y;
209     x = self.x = !self.canScrollHorizontal() ? 0 : x;
210     y = self.y = !self.canScrollVertical() ? 0 : y;
211     
212     if (!self.isScrolling && ((self.canScrollHorizontal() && deltaX !== 0) ||
213         (self.canScrollVertical() && deltaY !== 0))) scrollStart();
214     
215     content.translate(x, y);
216     
217     if (self.canScrollHorizontal() && deltaX !== 0) horizontalScrollBar.update(true);
218     if (self.canScrollVertical() && deltaY !== 0) verticalScrollBar.update(true);
219     
220     _mouseWheelTimeout = window.setTimeout(function() {
221       scrollEnd();
222     }, kAccelerationTimeout);
223     
224     evt.preventDefault();
225   });
226   
227   $element.bind('mousedown touchstart', function(evt) {
228     var scrollViewSize = self.getSize();
229     var contentSize = content.getSize();
230     
231     self.minimumX = scrollViewSize.width - contentSize.width;
232     self.minimumY = scrollViewSize.height - contentSize.height;
233     
234     _isDragging = true;
235     _startAccelerateX = self.x;
236     _startAccelerateY = self.y;
237     _lastTimeStamp = evt.timeStamp;
238     
239     if (evt.type === 'touchstart') {
240       var targetTouches = evt.targetTouches || evt.originalEvent.targetTouches;
241       _lastMouseX = targetTouches[0].pageX;
242       _lastMouseY = targetTouches[0].pageY;
243       _lastTouchIdentifier = targetTouches[0].identifier;
244     } else {
245       _lastMouseX = evt.pageX;
246       _lastMouseY = evt.pageY;
247     }
248     
249     stopDeceleration();
250     
251     if (self.canScrollHorizontal()) horizontalScrollBar.update(true);
252     if (self.canScrollVertical()) verticalScrollBar.update(true);
253     
254     $window.bind('mousemove touchmove', mouseMoveHandler);
255     $window.bind('mouseup touchend', mouseUpHandler);
256   });
257   
258   var mouseMoveHandler = function(evt) {
259     if (!_isDragging) return;
260     
261     var mouseX, mouseY;
262     
263     if (evt.type === 'touchmove') {
264       var touches = evt.touches || evt.originalEvent.touches;
265       var touch = touches[0];
266       
267       for (var i = 0, length = touches.length; i < length; i++) {
268         if (touches[i].identifier === _lastTouchIdentifier) {
269           touch = touches[i];
270           break;
271         }
272       }
273       
274       mouseX = touch.pageX;
275       mouseY = touch.pageY;
276     } else {
277       mouseX = evt.pageX;
278       mouseY = evt.pageY;
279     }
280     
281     var deltaX = mouseX - _lastMouseX;
282     var deltaY = mouseY - _lastMouseY;
283     var x = self.x + deltaX;
284     var y = self.y + deltaY;
285     
286     x -= ((x < self.minimumX) ? deltaX : (x > 0) ? deltaX : 0) / 2;
287     y -= ((y < self.minimumY) ? deltaY : (y > 0) ? deltaY : 0) / 2;
288     x = self.x = !self.canScrollHorizontal() ? 0 : x;
289     y = self.y = !self.canScrollVertical() ? 0 : y;
290     
291     content.translate(x, y);
292     
293     var accelerationTime = evt.timeStamp - _lastTimeStamp;
294     
295     if (accelerationTime > kAccelerationTimeout) {
296       _startAccelerateX = x;
297       _startAccelerateY = y;
298       _lastTimeStamp = evt.timeStamp;
299     }
300     
301     if (self.canScrollHorizontal() && deltaX !== 0) horizontalScrollBar.update();
302     if (self.canScrollVertical() && deltaY !== 0) verticalScrollBar.update();
303     if (!self.isScrolling && ((self.canScrollHorizontal() && deltaX !== 0) ||
304         (self.canScrollVertical() && deltaY !== 0))) scrollStart();
305     
306     _lastMouseX = mouseX;
307     _lastMouseY = mouseY;
308     
309     evt.preventDefault();
310   };
311   
312   var mouseUpHandler = function(evt) {
313     if (!_isDragging) return;
314     
315     var pagingEnabled = self.pagingEnabled;
316     var timeStamp = evt.timeStamp;
317     var accelerationTime = timeStamp - _lastTimeStamp;
318     var x = Math.min(Math.max(self.minimumX, self.x), 0);
319     var y = Math.min(Math.max(self.minimumY, self.y), 0);
320     
321     var bounce = function() {
322       $element.bind('webkitTransitionEnd transitionend MSTransitionEnd oTransitionEnd transitionEnd', bounceTransitionEndHandler);
323       content.translate(x, y, self.bounceTransitionDuration);
324       
325       if (self.canScrollHorizontal()) horizontalScrollBar.hide();
326       if (self.canScrollVertical()) verticalScrollBar.hide();
327       
328       self.x = _startAccelerateX = !self.canScrollHorizontal() ? 0 : x;
329       self.y = _startAccelerateY = !self.canScrollVertical() ? 0 : y;
330     };
331     
332     if (pagingEnabled) {
333       var acceleration = (timeStamp - _lastTimeStamp) / kAcceleration;
334       var accelerateDeltaX = self.x - _startAccelerateX;
335       var velocityX = accelerateDeltaX / acceleration;      
336       var pageWidth = $window.width();
337       var pageIndex = Math.round(x / pageWidth);
338       var pageCount = Math.floor(self.content.getSize().width / pageWidth);
339       var previousPage = self.currentPage;
340       var currentPage = Math.abs(pageIndex);
341       
342       if (previousPage === currentPage && Math.abs(velocityX) > kMinimumPageTurnVelocity) currentPage += (velocityX > 0) ? -1 : 1;
343       
344       currentPage = Math.min(Math.max(currentPage, 0), pageCount - 1);
345       x = -currentPage * pageWidth;
346       self.currentPage = currentPage;
347       
348       bounce();
349       
350       $element.trigger(SKScrollEventType.PageChanged);
351     }
352     
353     else {
354       if (accelerationTime < kAccelerationTimeout) {
355         if (x !== self.x || y !== self.y) {
356           bounce();
357         } else {
358           startDeceleration(timeStamp);
359         }
360       } else if (x !== self.x || y !== self.y) {
361         bounce();
362       } else {
363         scrollEnd();
364       }
365     }
366     
367     _isDragging = false;
368     _lastMouseX = -1;
369     _lastMouseY = -1;
370     _lastTimeStamp = timeStamp;
371     
372     $window.unbind('mousemove touchmove', mouseMoveHandler);
373     $window.unbind('mouseup touchend', mouseUpHandler);
374   };
375 };

485 var SKScrollBar = function(scrollView, type) {
486   this.scrollView = scrollView;
487   this.type = type;
488   
489   var $element = this.$element = $('<div class="sk-scroll-bar"/>');
490   var element = this.element = $element.get(0);

495     this.thickness = parseInt($element.css('height'), 10);
496     $element.css({ 'bottom': '0', 'left': '0' });

498     this.thickness = parseInt($element.css('width'), 10);

503 SKScrollBar.prototype = {
504   _size: 0,
505   _scrollViewSize: null,
506   _contentSize: null,
507   element: null,
508   $element: null,
509   scrollView: null,
510   type: SKScrollBarType.Horizontal,
511   minimumSize: 12,
512   thickness: 0,
513   setSize: function(value) {
514     this._size = value;
515     this.$element.css((this.type === SKScrollBarType.Horizontal) ? 'height' : 'width', this.thickness + 'px');
516     this.$element.css((this.type === SKScrollBarType.Horizontal) ? 'width' : 'height', value + 'px');
517   },
518   update: function(recalculateSize) {
519     var scrollView = this.scrollView;
520     var minimumSize = this.minimumSize;
521     var thickness = this.thickness;
522     var type = this.type;
523     var content = scrollView.content;
524     var scrollViewSize, contentSize
525     
526     if (recalculateSize) {
527       scrollViewSize = this._scrollViewSize = scrollView.getSize();
528       contentSize = this._contentSize = content.getSize();
529     } else {
530       scrollViewSize = this._scrollViewSize || scrollView.getSize();
531       contentSize = this._contentSize || content.getSize();
532     }
533     
534     var canScrollHorizontal = scrollView.canScrollHorizontal();
535     var canScrollVertical = scrollView.canScrollVertical();
536     var margin, size, scrollPosition, minimumPosition, position, translate3d, translate;
537     
538     if (type === SKScrollBarType.Horizontal) {
539       margin = (canScrollVertical ? thickness * 2 : thickness) + 1;
540       scrollPosition = scrollView.x;
541       size = (!recalculateSize && minimumPosition < scrollPosition && scrollPosition < 0) ?
542         this._size : Math.max(minimumSize, Math.round(
543           (scrollViewSize.width / contentSize.width) * (scrollViewSize.width - margin)
544         ));
545       
546       minimumPosition = scrollView.minimumX;
547       position = (scrollPosition / minimumPosition) * (scrollViewSize.width - margin - size);
548       
549       if (scrollPosition > 0) {
550         size = Math.round(Math.max(size - scrollPosition, thickness));
551         position = 1;
552       } else if (scrollPosition < minimumPosition) {
553         size = Math.round(Math.max(size - minimumPosition + scrollPosition, thickness));
554         position = scrollViewSize.width - size - margin;
555       }
556       
557       translate3d = 'translate3d(' + position + 'px, 0, 0)';
558       translate = 'translate(' + position + 'px, 0)';
559     } else {
560       margin = (canScrollHorizontal ? thickness * 2 : thickness) + 1;
561       scrollPosition = scrollView.y;
562       size = (!recalculateSize && minimumPosition < scrollPosition && scrollPosition < 0) ?
563         this._size : Math.max(minimumSize, Math.round(
564           (scrollViewSize.height / contentSize.height) * (scrollViewSize.height - margin)
565         ));
566       
567       minimumPosition = scrollView.minimumY;
568       position = (scrollPosition / minimumPosition) * (scrollViewSize.height - margin - size);
569       
570       if (scrollPosition > 0) {
571         size = Math.round(Math.max(size - scrollPosition, thickness));
572         position = 1;
573       } else if (scrollPosition < minimumPosition) {
574         size = Math.round(Math.max(size - minimumPosition + scrollPosition, thickness));
575         position = scrollViewSize.height - size - margin;
576       }
577       
578       translate3d = 'translate3d(0, ' + position + 'px, 0)';
579       translate = 'translate(0, ' + position + 'px)';
580     }
581     
582     if (this._size !== size) this.setSize(size);
583     
584     this.$element.css({
585       '-webkit-transform': translate3d,
586       '-moz-transform': translate,
587       '-ms-transform': translate,
588       '-o-transform': translate,
589       'transform': translate
590     });
591   },
592   show: function() {
593     var scrollView = this.scrollView;
594     var type = this.type;
595     
596     if ((!scrollView.showsHorizontalScrollIndicator && type === SKScrollBarType.Horizontal) ||
597         (!scrollView.showsVerticalScrollIndicator && type === SKScrollBarType.Vertical)) {
598       this.$element.removeClass('active');
599       return;
600     }
601     
602     this.$element.addClass('active');
603   },
604   hide: function() {
605     this.$element.removeClass('active');
606   }
607 };
