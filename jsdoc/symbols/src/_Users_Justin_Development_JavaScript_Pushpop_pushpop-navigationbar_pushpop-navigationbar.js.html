  2 
  3 // The base Pushpop object.
  4 var Pushpop = window['Pushpop'] || {};
  5 
  6 Pushpop.NavigationBar = function(element) {    
  7   var $element = this.$element = $(element);
  8   element = this.element = $element[0];
  9   
 10   var navigationBar = element.navigationBar;
 11   if (navigationBar) return navigationBar;
 12   
 13   element.navigationBar = this;
 14   
 15   var self = this;
 16 
 17   var viewStack = this.viewStack = Pushpop.getViewStackForElement(element);
 18   if (!viewStack) return;
 19   
 20   var tapToTop = $element.attr('data-tap-to-top') || 'false';
 21   tapToTop = this.tapToTop = this.tapToTop || (tapToTop !== 'false');
 22   
 23   var $tapToTopElement = this.$tapToTopElement = $('<div class="pp-navigationbar-tap-to-top"/>').appendTo($element);
 24   $tapToTopElement.bind('click', function(evt) {
 25     if (!self.tapToTop) return;
 26     
 27     var activeView = viewStack.getActiveView();
 28     var scrollView = activeView.element.scrollView;
 29     if (scrollView && !scrollView.isScrolling) scrollView.scrollToTop(true);
 30   });
 31   
 32   var $backButtonElement = this.$backButtonElement = $('<a class="pp-barbutton pp-barbutton-align-left pp-barbutton-style-back" href="#">Back</a>').appendTo($element);
 33   
 34   viewStack.$element.bind(Pushpop.EventType.WillPresentView, function(evt) {
 35     var view = evt.view;
 36     self.setTitle(view.title);
 37     self.loadNavbarButtons(view, evt.action)
 38   });
 39   
 40   // Prevent dragging of the Navigation Bar.
 41   $element.bind('touchmove', function(evt) { return false; });
 42   
 43   var activeView = viewStack.getActiveView();
 44   if (!activeView) return;
 45   
 46   this.setTitle(activeView.title);
 47   this.loadNavbarButtons(activeView);
 48 };
 49 
 50 Pushpop.NavigationBar.prototype = {
 51   element: null,
 52   $element: null,
 53   $tapToTopElement: null,
 54   $backButtonElement: null,
 55   $viewSpecificNavButtons: null,
 56   $titleElement: null,
 57   viewStack: null,
 58   tapToTop: false,
 59   setTitle: function(value) {
 60     var $titleElement = this.$titleElement;
 61     if (!$titleElement) $titleElement = this.$titleElement = $('<h1 class="pp-navigationbar-title"/>').appendTo(this.$element);
 62     
 63     $titleElement.html(value || '');
 64   },
 65 	loadNavbarButtons: function(view, action) {
 66     if (action === 'parent-push' || action === 'parent-pop') return;
 67     
 68 	  // Clear any nav buttons currently in the navbar (other than the back button)
 69 	  if (this.$viewSpecificNavButtons) this.$viewSpecificNavButtons.remove();
 70 
 71     // Append buttons for this view
 72     this.$element.append(this.$viewSpecificNavButtons = view.$navbarButtons);
 73     
 74     // Show/Hide the back button
 75     switch (action) {
 76       case 'push':
 77         if (!view.hideNavBackButton) {
 78           this.$backButtonElement.addClass('pop active');
 79         } else {
 80           this.$backButtonElement.removeClass('pop active');
 81         }
 82         break;
 83       case 'pop':
 84         if (!view.hideNavBackButton && this.viewStack.views.length > 1) {
 85           this.$backButtonElement.addClass('pop active');
 86         } else {
 87           this.$backButtonElement.removeClass('pop active');
 88         }
 89       default:
 90         break;
 91     }
 92 	}
 93 };
 94 
 95 $(function() {
 96   var $navigationBars = $('.pp-navigationbar');
 97   $navigationBars.each(function(index, element) {
 98     new Pushpop.NavigationBar(element);
 99   });
