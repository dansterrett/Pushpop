259   Pushpop.TableViewPickerCell = function(element) {
260     var $element = this.$element = $(element);
261

275     var $viewElement = $('<div class="pp-view sk-scroll-view pp-tableview-picker-view" id="pp-tableview-picker-view-' + (++_lastPickerViewId) + '"/>');

284     
285     var $selectedTextElement = this.$selectedTextElement = $('<span/>').appendTo($element);

288

290     
291     if (isMultiple) {
292       for (var i = 0, length = values.length; i < length; i++) this.setValue(values[i]);
293     } else {
294       if (values.length > 0) this.setValue(values[0]);
295     }

297 		// This string is used to store the value of each item that is selected as the user
298 		//   is drilling down through the layers of the tableview data.
299 		var valueHierarchy = '';
300     
301 		var callbackForDidSelectCell = function(evt) {
302       var $cellElement = evt.$cellElement;
303       
304 		  // Check if the cell was disabled or a header cell.  If so, return.
305 		  if ($cellElement.hasClass('pp-tableview-cell-disabled') || $cellElement.hasClass('header')) return;
306 		  
307       // Ensure that the list is selectable. Once an item is selected, the list is marked as unselectable
308       // until the next time its view is presented
309       var parentList = $cellElement.parent();
310       if (parentList.hasClass('pp-tableview-unselectable')) return;
311       
312       parentList.addClass('pp-tableview-unselectable');
313 		  		  
314       var value = $cellElement.data('value');
315       if (!value) return;
316       
317 			// Is value an array?
318 			if ($.isArray(value)) {
319 				// Add the id of the item to the value hierarchy
320 				valueHierarchy += (valueHierarchy.length > 0 ? self.valuesDelimiter : '') + $cellElement.data('id');
321 				// Create a new pp-tableview with the choices being the values in this array
322 				var $viewElement = $('<div class="pp-view sk-scroll-view temp-view pp-tableview-picker-view" id="pp-tableview-picker-view-' + (++_lastPickerViewId) + '"/>');
323 				// Append the new tableview to the viewstack
324 		    viewStack.$element.append($viewElement);
325 		    // Bind the DidPopView event on the view element
326 		    $viewElement.bind(Pushpop.EventType.DidPopView, callbackForDidPopView);
327 				// Give the new tableview a scrollview
328 				var scrollView = new SKScrollView($viewElement);
329 		    var $scrollViewContentElement = scrollView.content.$element;
330 				var view = new Pushpop.View($viewElement);
331 				// Make a new ul for the items
332 				var $ul = $('<ul class="pp-tableview pp-tableview-unselectable" />');
333 				// Add the header item.  Give it the same text as the previous view
334 				$ul.append('<li class="header">' + $cellElement.siblings(':first-child').html() + '</li>')
335 				// Add the root level items. Note: we only add the root level items 
336 				// so we don't load the dom with too many elements
337 				for (var i = 0, length = value.length; i < length; i++) {
338 					// Create a new list item for this item
339 					var $li = $('<li data-id="' + value[i].id + '">' + value[i].title + '</li>');
340 					
341 					// If this value is not an array, check to see if it is already selected
342 					if (!$.isArray(value[i].value) && (self._value.indexOf(valueHierarchy + self.valuesDelimiter + value[i].value) > -1)) {
343 						$li.addClass('pp-tableview-accessory-checkmark pp-tableview-cell-disabled');
344 					}
345 					
346 					$li.data('value', value[i].value);
347 					
348 					// Append the list item to the list
349 					$ul.append($li);
350 				}
351 				
352 				var tableView = new Pushpop.TableView($ul);
353 		
354 				// Append the list to the scrollViewContentElement
355 				$scrollViewContentElement.append($ul);
356 				
357 				// Bind the DidSelectCell event for this tableViewElement
358 				$ul.bind(Pushpop.EventType.DidSelectCell, callbackForDidSelectCell);
359         $viewElement.bind(Pushpop.EventType.DidPresentView, callbackForDidPresentView);
360 
361 				// Push the table view
362 				viewStack.push(view);
363 			} else {
364 				// Add this value to the value hierarchy
365 				valueHierarchy += (valueHierarchy.length > 0 ? self.valuesDelimiter : '') + value;
366       	var text = self.getTextByValue(valueHierarchy);
367       
368 	      if (self.isMultiple) {
369 	        self.setValue(valueHierarchy);
370 	        $element.trigger($.Event(Pushpop.EventType.DidAddValue, {
371 	          cellElement: element,
372 	          $cellElement: $element,
373 	          value: valueHierarchy,
374 	          text: text
375 	        }));
376 	      } else {
377 	        self.setValue(valueHierarchy);
378 	      }
379       
380 	      $element.trigger($.Event(Pushpop.EventType.DidChangeValue, {
381 	        cellElement: element,
382 	        $cellElement: $element,
383 	        value: self.getValue()
384 	      }));
385 	
386 				// Clear the value hierarchy for next time
387 				valueHierarchy = '';
388       
389 				// After popping the parent view, remove all temporary views from the dom
390 	      viewStack.pop(self.getParentTableView().getView(), function() { $('.temp-view').remove(); });
391 			}
392     };
393     
394     var callbackForDidPopView = function(evt) {
395       // Remove the last value off the valueHierarchry
396       var lastIndexOfDelimiter = valueHierarchy.lastIndexOf(self.valuesDelimiter);
397       valueHierarchy =  ((lastIndexOfDelimiter > 0) ? valueHierarchy.substring(0, lastIndexOfDelimiter) : '');
398       evt.view.$element.remove();
399     };
400     
401     var callbackForDidPresentView = function(evt) {
402       // Remove the pp-tableview-unselectable class from the unordered list, so its items can be selected.
403       // This is necessary because we temporarily "disable" selecting the items from both the current view 
404       // and the new view between the time an item is selected and the time the new view has finished being pushed.
405       evt.view.$element.find('ul.pp-tableview-unselectable').removeClass('pp-tableview-unselectable');
406     };
407     
408     $tableViewElement.bind(Pushpop.EventType.DidSelectCell, callbackForDidSelectCell);
409     $viewElement.bind(Pushpop.EventType.DidPresentView, callbackForDidPresentView);
410   };

412   Pushpop.TableViewPickerCell.prototype = {
413     _value: null,
414 		_dataSource: null,
415     element: null,
416     $element: null,
417     $selectedTextElement: null,
418     view: null,
419     tableView: null,
420     isMultiple: false,
421 		valuesDelimiter: '-',
422 		displayTextProperty: 'title',
423     getParentTableView: function() {
424       return this.$element.parents('.pp-tableview').first().data('tableview');
425     },
426     getTextByValue: function(value) {
427       // Does this picker cell have a datasource?
428 			if (this.getDataSource()) {
429 			  var item = this.getDataSourceItemByValue(value);
430 			  return (item ? item[this.displayTextProperty] : '');
431 			} 
432 			// There's no need to drill down through the data
433 			else return this.tableView.$element.children('[data-value="' + value + '"]').first().text();
434     },
435     getDataSourceItemByValue: function(value) {
436       var valuesArray = [value];
437       // Do we need to drill down?
438       if (('' + value).indexOf(this.valuesDelimiter) > -1) {
439         valuesArray = value.split(this.valuesDelimiter);
440       }
441 			  
442 			return this.getDataSourceItemByValueRecursively(valuesArray, this.getDataSource());
443 	    
444     },
445 		// Recursive method to drill down through the data (if necessary), 
446 		// and return the object with the specified value
447 		getDataSourceItemByValueRecursively: function(valuesArray, arrayOfItems) {
448 			// Get the first value in the array. Then remove it from the list
449 			var valueToSearchFor = valuesArray[0];
450 			valuesArray.splice(0, 1);
451 			
452 			// If this is not the last level, then we are need to search on the id property of the item,
453 			//   otherwise, we need to search on the value property of the item.
454 			var propertyToSearch = (valuesArray.length > 0 ? 'id' : 'value');
455 
456 			// Loop through arrayOfItems and find the item with the right value
457 			for(var i = 0, length = arrayOfItems.length; i < length; i++) {
458 				if (arrayOfItems[i][propertyToSearch] == valueToSearchFor) {
459 					// Are there more levels to drill down to?
460 					if (valuesArray.length > 0) return this.getDataSourceItemByValueRecursively(valuesArray, arrayOfItems[i].value);
461 					// This is the last level, so return the text
462 					else return arrayOfItems[i];
463 				}
464 			}
465 		},
466     getValue: function() {
467       return this._value;
468     },
469     setValue: function(value, text) {
470       var element = this.element;
471       var $element = this.$element;
472       var $tableViewElement = this.tableView.$element;
473       var isMultiple = this.isMultiple;
474       text = text || this.getTextByValue(value);
475       
476       if (isMultiple) {
477         for (var i = 0, length = this._value.length; i < length; i++) if (this._value[i] == value) return;
478         
479         this._value.push(value);
480         $element.attr('data-value', this._value.join(',')).data('value', this._value);
481         this.$selectedTextElement.html(null);
482         
483         var $valueCellElement = $('<li class="pp-tableview-picker-value-cell pp-tableview-editing-cell pp-tableview-editing-accessory-delete" data-value="' + value + '">' + text + '</li>');
484         $valueCellElement.data('pickerCell', this);
485         $element.before($valueCellElement);
486         $tableViewElement.children('[data-value="' + value + '"]').first().addClass('pp-tableview-accessory-checkmark pp-tableview-cell-disabled');
487       } else {
488         this._value = [value];
489         $element.attr('data-value', this._value.join(',')).data('value', this._value);
490         this.$selectedTextElement.html(text);
491         
492         $tableViewElement.children('.pp-tableview-accessory-checkmark').removeClass('pp-tableview-accessory-checkmark pp-tableview-cell-disabled');
493         $tableViewElement.children('[data-value="' + value + '"]').first().addClass('pp-tableview-accessory-checkmark pp-tableview-cell-disabled');
494       }
495     },
496     getDataSource: function() {
497 			return this._dataSource;
498 		},
499 		setDataSource: function(value) {
500 			this._dataSource = value;
501 			
502 			// Get the existing tableview that holds the available choices
503 			var $ul = this.tableView.$element;
504 			
505 			// Clear any existing items (except the header)
506 			$ul.children(':not(:first-child)').remove();
507 			
508 			// Add the root level items. Note: we only add the root level items 
509 			// so we don't load the dom with too many elements
510 			for (var i = 0, length = value.length; i < length; i++) {
511 				// Create a new list item for this item
512 				var $li = $('<li data-id="' + value[i].id + '">' + value[i].title + '</li>');
513 				// Set the items value.  This could be a simple type (int or string) or an array of objects.
514 				if ($.isArray(value[i].value)) {
515 					$li.data('value', value[i].value);
516 				} else {
517 					// Set the "data-value" attribute, because there is some code looking at this attribute to 
518 					//   get the text and to add the checkmark to the item, rather than using the .data() method.
519 					$li.attr('data-value', value[i].value);
520 				}
521 				
522 				// Append the list item to the list
523 				$ul.append($li);
524 			}			
525 		},
526 		removeValue: function(value) {
527       var element = this.element;
528       var $element = this.$element;
529       var isMultiple = this.isMultiple;
530       var text = this.getTextByValue(value);
531       
532       if (isMultiple) {
533         var index;
534         
535         for (var i = 0, length = this._value.length; i < length; i++) {
536           if (this._value[i] == value) {
537             index = i;
538             break;
539           }
540         }
541         
542         if (index === undefined) return;
543         
544         this._value.splice(index, 1);
545         
546         var $tableViewElement = this.tableView.$element;
547         $tableViewElement.children('[data-value="' + value + '"]').first().removeClass('pp-tableview-accessory-checkmark pp-tableview-cell-disabled');
548         $element.prevAll('[data-value="' + value + '"]').first().remove();
549       } else {
550         this._value = [];
551       }
552       
553       $element.attr('data-value', this._value.join(',')).data('value', this._value);
554       this.$selectedTextElement.html(null);
555     },
556     removeAll: function() {
557       var i, length, values = [];
558       
559       for (i = 0, length = this._value.length; i < length; i++) values.push(this._value[i]);
560       for (i = 0, length = values.length; i < length; i++) this.removeValue(values[i]);
561     },
562     show: function() {
563       var view = this.view;
564       var viewStack = view.getViewStack();
565       
566       viewStack.push(view);
567     }
