  6 /**
  7   Creates a new TableView.
  8   @param {HTMLUListElement} element The <ul/> element to initialize as a new TableView.
  9   @constructor
 10 */
 11 Pushpop.TableView = function TableView(element) {
 12   var $window = $(window['addEventListener'] ? window : document.body);
 13   
 14   var $element = this.$element = $(element);
 15   var element = this.element = $element[0];
 16   
 17   var tableView = element.tableView;
 18   if (tableView) return tableView;
 19   
 20   element.tableView = this;
 21   
 22   var scrollViewElement = $element.parents('.sk-scroll-view')[0];
 23   if (!scrollViewElement) return;
 24   
 25   var self = this;
 26   
 27   var reusableCells = this._reusableCells = {};
 28   var visibleCells = this._visibleCells = [];
 29   var selectedRowIndexes = this._selectedRowIndexes = [];
 30   
 31   var scrollView = this.scrollView = scrollViewElement.scrollView;
 32   
 33   var containsSearchBar = $element.attr('data-contains-search-bar') || 'false';
 34   containsSearchBar = containsSearchBar !== 'false';
 35   
 36   var searchBar = null;
 37   if (containsSearchBar) searchBar = this._searchBar = new Pushpop.TableViewSearchBar(this);
 38   
 39   var numberOfBufferedCells = this._numberOfBufferedCells;
 40   var selectionTimeoutDuration = this._selectionTimeoutDuration;
 41   var lastOffset = -scrollView.y;
 42   var topMargin = window.parseInt($element.css('margin-top'), 10);
 43   
 44   // Render table view cells "virtually" when the view is scrolled.
 45   scrollView.$element.bind(SKScrollEventType.ScrollChange, function(evt) {
 46     var offset = -scrollView.y;
 47     if (offset < 0) return;
 48     
 49     var firstCellElement = $element.children('li:first-child')[0];
 50     var lastCellElement = $element.children('li:last-child')[0];
 51     if (!firstCellElement || !lastCellElement || firstCellElement === lastCellElement) return;
 52     
 53     var dataSource = self.getDataSource();
 54     var rowHeight = self.getRowHeight();
 55     var numberOfRows = self._numberOfRows;
 56     var visibleCellCount = self.getCalculatedNumberOfVisibleCells();
 57     var selectedRowIndex = self.getIndexForSelectedRow();
 58     
 59     var firstCell = firstCellElement.tableViewCell;
 60     var firstCellIndex = firstCell.getIndex();
 61     var lastCellIndex = firstCellIndex + visibleCellCount - 1;
 62     
 63     // Manually calculate offset instead of calling .offset().
 64     var margin = scrollView.getMargin();
 65     var firstCellOffset = margin.top - offset;
 66     var lastCellOffset = firstCellOffset + (visibleCellCount * rowHeight);
 67     var delta = offset - lastOffset;
 68     var visibleHeight = self.getVisibleHeight();
 69     
 70     lastOffset = offset;
 71     
 72     var newMarginTopDelta = 0;
 73     var newMarginBottomDelta = 0;
 74     
 75     // Handle scrolling when swiping up (scrolling towards the bottom).
 76     if (delta > 0 && lastCellIndex + 1 < numberOfRows && firstCellOffset < 0 - (rowHeight * numberOfBufferedCells)) {
 77       $element.children('li:nth-child(-n+' + numberOfBufferedCells + ')').each(function(index, element) {
 78         var newCellIndex = lastCellIndex + index + 1;
 79         if (newCellIndex >= numberOfRows) return;
 80         
 81         var cell = element.tableViewCell;
 82         cell.prepareForReuse();
 83         
 84         var newCell = dataSource.getCellForRowAtIndex(self, newCellIndex);
 85         if (self.isRowSelectedAtIndex(newCellIndex)) newCell.setSelected(true);
 86         $element.append(newCell.$element);
 87         
 88         newMarginTopDelta += rowHeight;
 89         newMarginBottomDelta -= rowHeight;
 90       });
 91       
 92       scrollView.setMargin({
 93         top: margin.top + newMarginTopDelta,
 94         bottom: margin.bottom + newMarginBottomDelta
 95       });
 96     }
 97     
 98     // Handle scrolling when swiping down (scrolling towards the top).
 99     else if (delta < 0 && firstCellIndex - 1 >= 0 && lastCellOffset > visibleHeight + (rowHeight * numberOfBufferedCells)) {
100       $element.children('li:nth-child(n+' + (visibleCellCount - numberOfBufferedCells + 1) + ')').each(function(index, element) {
101         var newCellIndex = firstCellIndex - index - 1;
102         if (newCellIndex < 0) return;
103         
104         var cell = element.tableViewCell;
105         cell.prepareForReuse();
106         
107         var newCell = dataSource.getCellForRowAtIndex(self, newCellIndex);
108         if (self.isRowSelectedAtIndex(newCellIndex)) newCell.setSelected(true);
109         $element.prepend(newCell.$element);
110         
111         newMarginTopDelta -= rowHeight;
112         newMarginBottomDelta += rowHeight;
113       });
114       
115       scrollView.setMargin({
116         top: margin.top + newMarginTopDelta,
117         bottom: margin.bottom + newMarginBottomDelta
118       });
119     }
120   });
121   
122   // Handle case when table view is scrolled to the top (e.g.: tapping top of navigation bar).
123   scrollView.$element.bind(SKScrollEventType.WillScrollToTop, function(evt) {    
124     var firstCellElement = $element.children('li:first-child')[0];
125     if (firstCellElement) firstCellElement.tableViewCell.setIndex(0);
126   }).bind(SKScrollEventType.DidScrollToTop, function(evt) { self.reloadData(); });
127   
128   // Handle mouse/touch events to allow the user to tap accessory buttons.
129   var isPendingAccessoryButtonTap = false;
130 
131   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchstart' in window) ? 'touchstart' : 'mousedown', function(evt) {
132     isPendingAccessoryButtonTap = true;
133   });
134   
135   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchmove' in window) ? 'touchmove' : 'mousemove', function(evt) {
136     if (!isPendingAccessoryButtonTap) return;
137     isPendingAccessoryButtonTap = false;
138   });
139   
140   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchend' in window) ? 'touchend' : 'mouseup', function(evt) {
141     if (!isPendingAccessoryButtonTap) return;
142     isPendingAccessoryButtonTap = false;
143     
144     var tableViewCell = $(this).parent()[0].tableViewCell;
145     if (!tableViewCell) return;
146     
147     $element.trigger($.Event(Pushpop.TableView.EventType.AccessoryButtonTappedForRowWithIndex, {
148       tableView: self,
149       index: tableViewCell.getIndex()
150     }));
151   });
152   
153   // Handle mouse/touch events to allow the user to make row selections.
154   var isPendingSelection = false, selectionTimeout = null;
155 
156   $element.delegate('li', !!('ontouchstart' in window) ? 'touchstart' : 'mousedown', function(evt) {
157     
158     // Don't allow row to be selected if an accessory button is pending a tap.
159     if (isPendingAccessoryButtonTap) return;
160     
161     isPendingSelection = true;
162     
163     var tableViewCell = this.tableViewCell;
164     
165     selectionTimeout = window.setTimeout(function() {
166       if (!isPendingSelection) return;
167       isPendingSelection = false;
168       
169       self.deselectAllRows();
170       self.selectRowAtIndex(tableViewCell.getIndex());
171     }, selectionTimeoutDuration);
172   });
173   
174   $element.bind(!!('ontouchmove' in window) ? 'touchmove' : 'mousemove', function(evt) {
175     if (!isPendingSelection) return;
176     isPendingSelection = false;
177     
178     window.clearTimeout(selectionTimeout);
179   });
180   
181   $element.delegate('li', !!('ontouchend' in window) ? 'touchend' : 'mouseup', function(evt) {
182     if (!isPendingSelection) return;
183     isPendingSelection = false;
184     
185     window.clearTimeout(selectionTimeout);
186   
187     var tableViewCell = this.tableViewCell;
188     
189     self.deselectAllRows();
190     self.selectRowAtIndex(tableViewCell.getIndex());
191   });
192   
193   // Create a new data source from a data set URL.
194   var dataSetUrl = $element.attr('data-set-url');
195   if (dataSetUrl) {
196     $.getJSON(dataSetUrl, function(dataSet) {
197       $element.html(null);
198       self.setDataSource(new Pushpop.TableViewDataSource(dataSet));
199     });
200   }
201   
202   // Create a new data source from existing <li/> elements.
203   else {
204     (function(self, $element) {
205       var dataSet = [];
206       
207       var dashAlpha = /-([a-z]|[0-9])/ig;
208       var camelCase = function(string) { return string.replace(dashAlpha, function(all, letter) { return (letter + '').toUpperCase(); }); };
209       
210       $element.children('li').each(function(index, element) {
211         var data = { title: $(element).html() };
212         
213         var attributes = element.attributes;
214         var attribute, attributeName;
215         for (var i = 0, length = attributes.length; i < length; i++) {
216           attribute = attributes[i];
217           attributeName = attribute.name;
218           
219           if (attributeName.indexOf('data-') === 0) data[camelCase(attributeName.substring(5))] = attribute.value;
220         }
221         
222         if (!data['reuseIdentifier']) data.reuseIdentifier = 'pp-table-view-cell-default';
223       
224         dataSet.push(data);
225       });
226       
227       $element.html(null);
228       self.setDataSource(new Pushpop.TableViewDataSource(dataSet));
229     })(self, $element);
230   }
231 };

244   var reusableCellPrototypes = Pushpop.TableView._reusableCellPrototypes, items = [];
245   for (var reusableCellPrototype in reusableCellPrototypes) items.push(reusableCellPrototypes[reusableCellPrototype]);

251 Pushpop.TableView.registerReusableCellPrototype = function(cellPrototype) {
252   if (!cellPrototype) return;
253   
254   var reuseIdentifier = cellPrototype.getReuseIdentifier();
255   if (!reuseIdentifier) return;
256   
257   Pushpop.TableView._reusableCellPrototypes[reuseIdentifier] = cellPrototype;

275   
276   _reusableCells: null,
277   
278   /**
279     Returns an object containing arrays of reusable cells that are not currently
280     visible.
281     @description The object contains an array property for each "reuse identifier"
282     in use. For example, if there are two types of TableViewCells in this TableView
283     with two different reuse identifiers (cellType1 and cellType2), this object
284     would contain two arrays that can be accessed using square bracket notation
285     like this: reusableCells['cellType1'] or reusableCells['cellType2']
286     @type Object
287   */
288   getReusableCells: function() { return this._reusableCells; },
289   
290   /**
291     Returns a new or recycled TableViewCell with the specified reuse identifier.
292     @description This method will first attempt to reuse a recycled TableViewCell
293     with the specified reuse identifier. If no recycled TableViewCells with that
294     reuse identifier are available, a new one will be instantiated and returned.
295     The TableViewCell that is returned is always added to the |visibleCells| array.
296     @type Pushpop.TableViewCell
297   */
298   dequeueReusableCellWithIdentifier: function(reuseIdentifier) {
299     var visibleCells = this.getVisibleCells();
300     var reusableCells = this.getReusableCells();
301     reusableCells = reusableCells[reuseIdentifier] || (reusableCells[reuseIdentifier] = []);
302     
303     var cell = null, cellPrototype = null;
304     
305     if (reusableCells.length > 0) {
306       cell = reusableCells.pop();
307     } else {
308       cellPrototype = Pushpop.TableView.getReusableCellPrototypeWithIdentifier(reuseIdentifier);
309       cell = (cellPrototype) ? new cellPrototype.constructor(reuseIdentifier) : new Pushpop.TableViewCell(reuseIdentifier);
310     }
311     
312     visibleCells.push(cell);
313     cell.tableView = this;
314     
315     return cell;
316   },

365     var selectedRowIndexes = this._selectedRowIndexes;

370   /**
371     Selects the row at the specified index and optionally animates the selection if
372     the row is currently visible.
373     @description NOTE: This method will not modify any other existing selections.
374     @param {Number} index The index of a row in the data source to select.
375     @param {Boolean} [animated] A flag indicating if the selection should be animated
376     if the row is currently visible.
377   */
378   selectRowAtIndex: function(index, animated) {
379     var $element = this.$element;
380     this._selectedRowIndexes.push(index);
381     
382     var tableViewCell, $cells = this.$element.children();
383     for (var i = 0, length = $cells.length; i < length; i++) {
384       tableViewCell = $cells[i].tableViewCell;
385       if (tableViewCell.getIndex() === index) {
386         tableViewCell.setSelected(true);
387         break;
388       }
389     }
390     
391     $element.trigger($.Event(Pushpop.TableView.EventType.DidSelectRowAtIndex, {
392       tableView: this,
393       index: index
394     }));
395   },
396   
397   /**
398     De-selects the row at the specified index and optionally animates the de-selection
399     if the row is currently visible.
400     @description NOTE: This method will not modify any other existing selections.
401     @param {Number} index The index of a row in the data source to de-select.
402     @param {Boolean} [animated] A flag indicating if the de-selection should be
403     animated if the row is currently visible.
404   */
405   deselectRowAtIndex: function(index, animated) {
406     var $element = this.$element;
407     var selectedRowIndexes = this._selectedRowIndexes;
408     for (var i = 0, length = selectedRowIndexes.length; i < length; i++) {
409       if (selectedRowIndexes[i] === index) {
410         selectedRowIndexes.splice(i, 1);
411         break;
412       }
413     }
414     
415     var tableViewCell, $selectedCells = $element.children('.pp-table-view-selected-state');
416     for (var i = 0, length = $cells.length; i < length; i++) {
417       tableViewCell = $cells[i].tableViewCell;
418       if (tableViewCell.getIndex() === index) {
419         tableViewCell.setSelected(false);
420         break;
421       }
422     }
423     
424     $element.trigger($.Event(Pushpop.TableView.EventType.DidDeselectRowAtIndex, {
425       tableView: this,
426       index: index
427     }));
428   },

432   */

434     var $element = this.$element;
435     var selectedRowIndexes = this._selectedRowIndexes;
436     for (var i = 0, length = selectedRowIndexes.length; i < length; i++) {
437       $element.trigger($.Event(Pushpop.TableView.EventType.DidDeselectRowAtIndex, {
438         tableView: this,
439         index: selectedRowIndexes[i]

442     
443     selectedRowIndexes.length = 0;
444     
445     $element.children('.pp-table-view-selected-state').each(function(index, element) {
446       element.tableViewCell.setSelected(false);
447     });

454     var numberOfRows = this._numberOfRows;
455     var visibleCellCount = Math.min(this.getCalculatedNumberOfVisibleCells(), numberOfRows);

459

471     Reloads the data source for the TableView and resets the scroll position back
472     to the top of the TableView.
473     @description Typically this method is called if/when the data source changes or
474     if there is a change in the search string.
475   */
476   reloadData: function() {
477     var $element = this.$element;    
478     
479     var dataSource = this.getDataSource();
480     var visibleCells = this.getVisibleCells();
481     
482     var i, length, visibleCellsToReuse = [];
483     for (i = 0, length = visibleCells.length; i < length; i++) visibleCellsToReuse.push(visibleCells[i]);
484     for (i = 0, length = visibleCellsToReuse.length; i < length; i++) visibleCellsToReuse[i].prepareForReuse();
485     
486     var numberOfRows = this._numberOfRows = dataSource.getNumberOfRows();
487     var visibleCellCount = Math.min(this.getCalculatedNumberOfVisibleCells(), numberOfRows);
488     var hiddenCellCount = numberOfRows - visibleCellCount;
489     
490     for (i = 0; i < visibleCellCount; i++) {
491       var cell = dataSource.getCellForRowAtIndex(this, i);      
492       $element.append(cell.$element);
493     }
494     
495     this.resetScrollView();
496   },

568 /**
569   Creates a new data source for a TableView.
570   @param {Array} [dataSet] An optional array of data to initialize a default data source.
571   @param {Array} [dataSet.id] The unique identifier for a specific record.
572   @param {Array} [dataSet.value] The (sometimes) hidden value for a specific record.
573   @param {Array} [dataSet.title] The title to be displayed in a TableViewCell for a specific record.
574   @constructor
575 */
576 Pushpop.TableViewDataSource = function TableViewDataSource(dataSet) {
577   if (!dataSet || dataSet.constructor !== Array) return;
578   
579   this.setDataSet(dataSet);
580   
581   // Default implementation if using an in-memory data set.
582   this.getNumberOfRows = function() { return this.getFilteredDataSet().length; };
583   
584   // Default implementation if using an in-memory data set.
585   this.getCellForRowAtIndex = function(tableView, index) {
586     var data = this.getFilteredDataSet()[index];
587     var reuseIdentifier = data.reuseIdentifier || this.getDefaultReuseIdentifier();
588     var cell = tableView.dequeueReusableCellWithIdentifier(reuseIdentifier);
589     
590     cell.setIndex(index);
591     cell.setAccessoryType(data.accessoryType);
592     cell.setData(data);
593     
594     return cell;
595   };
596 };

598 /**
599   @description NOTE: In order to have a TableView with custom TableViewCells, a custom
600   TableViewDataSource must be implemented with at least the two required methods defined
601   in this prototype. A third optional method may also be implemented as defined in this
602   prototype.
603 */
604 Pushpop.TableViewDataSource.prototype = {
605   constructor: Pushpop.TableViewDataSource,
606   
607   /**
608     REQUIRED: Returns the number of rows provided by this data source.
609     @type Number
610   */
611   getNumberOfRows: function() { return 0; },
612   
613   /**
614     REQUIRED: Returns a TableViewCell for the specified index.
615     @param {Pushpop.TableView} tableView The TableView the TableViewCell should be returned for.
616     @param {Number} index The index of the data to be used when populating the TableViewCell.
617     @type Pushpop.TableViewCell
618   */
619   getCellForRowAtIndex: function(tableView, index) { return null; },
620   
621   /**
622     OPTIONAL: Determines if the table should be reloaded following a change in the search string.
623     @description The default implementation assumes that the data set is fully loaded into
624     memory and executes the current filter function against each item in the data set. If the
625     filtered data set has changed since the last reload, it will return |true| which will force
626     the associated TableViewSearchBar to reload the data for the TableView. In a custom data
627     source that does not use an in-memory data set (e.g.: WebSQL or HTML5 LocalStorage), it is
628     recommended to override this method to perform any necessary queries asynchronously and
629     immediately return |false|. Once the asynchronous queries have completed, the application
630     should then manually call .reloadData() on the TableView to force an update (See the WebSQL
631     demo for an example on this implementation).
632     @param {String} searchString The search string to be used for matching items in the data set.
633     @param {Boolean} [isCaseSensitive] An optional boolean flag for forcing a case-sensitive search.
634     @type Boolean
635   */
636   shouldReloadTableForSearchString: function(searchString, isCaseSensitive) {
637     var dataSet = this.getDataSet();
638     if (!dataSet) return false;
639     
640     var filterFunction = this.getFilterFunction();
641     var tableView = this.getTableView();
642     
643     if (!filterFunction || typeof filterFunction !== 'function' || !searchString) {
644       this._lastSearchString = null;
645       
646       if (this._filteredDataSet !== dataSet) {
647         this._filteredDataSet = dataSet;
648         return true;
649       }
650       
651       return false;
652     }
653     
654     var filteredDataSet = [];
655     var regExp = new RegExp(searchString + '+', (!isCaseSensitive ? 'i' : '') + 'm');
656     var item, i, length;
657     
658     // The search string is a continuation of the last search string (e.g.: 'ab' -> 'abc').
659     if (searchString.indexOf(this._lastSearchString) === 0) {
660       
661       // Search the previous filtered data set instead of the whole data set.
662       var lastFilteredDataSet = this._filteredDataSet;
663       for (i = 0, length = lastFilteredDataSet.length; i < length; i++) if (filterFunction(regExp, item = lastFilteredDataSet[i])) filteredDataSet.push(item);
664     }
665     
666     // The search string is NOT a contination of the last search string (e.g.: 'abc' -> 'ab').
667     else {
668       
669       // Search the whole data set.
670       for (i = 0, length = dataSet.length; i < length; i++) if (filterFunction(regExp, item = dataSet[i])) filteredDataSet.push(item);
671     }
672     
673     this._filteredDataSet = filteredDataSet;
674     this._lastSearchString = searchString;
675     return true;
676   },

770 Pushpop.TableViewSearchBar = function TableViewSearchBar(tableView) {
771   var $element = this.$element = $('<div class="pp-table-view-search-bar"/>');
772   var element = this.element = $element[0];
773   
774   element.tableViewSearchBar = this;
775   
776   var self = this;
777   
778   var $input = this.$input = $('<input type="text" placeholder="Search"/>').appendTo($element);
779   var $cancelButton = this.$cancelButton = $('<a class="pp-table-view-search-bar-button" href="#">Cancel</a>').appendTo($element);
780   var $clearButton = this.$clearButton = $('<a class="pp-table-view-search-bar-clear-button" href="#"/>').appendTo($element);
781   var $overlay = this.$overlay = $('<div class="pp-table-view-search-bar-overlay"/>');
782   
783   var willFocus = false;
784   
785   $input.bind('mousedown touchstart', function(evt) {
786     if ($input.is(':focus')) {
787       evt.stopPropagation();
788       return;
789     }
790     
791     evt.preventDefault();
792     willFocus = true;
793   });
794   $input.bind('mousemove touchmove', function(evt) { willFocus = false; });
795   $input.bind('mouseup touchend', function(evt) {
796     if ($input.is(':focus')) {
797       evt.stopPropagation();
798       return;
799     }
800     
801     evt.preventDefault();
802     if (willFocus) $input.trigger('focus');
803   });
804   $input.bind('focus', function(evt) {
805     if (!willFocus) {
806       $input.trigger('blur');
807       return false;
808     }
809     
810     willFocus = false;
811     self._tableView.resetScrollView();
812     
813     window.setTimeout(function() {
814       $overlay.addClass('pp-active');
815       if ($input.val()) $clearButton.addClass('pp-active');
816     }, 0);
817   });
818   $input.bind('blur', function(evt) { $overlay.removeClass('pp-active'); $clearButton.removeClass('pp-active'); });
819   $cancelButton.bind('mousedown touchstart', function(evt) { evt.stopPropagation(); evt.preventDefault(); });
820   $cancelButton.bind('mouseup touchend', function(evt) { $input.val(null).trigger('keyup').trigger('blur'); });
821   $clearButton.bind('mousedown touchstart', function(evt) { evt.stopPropagation(); evt.preventDefault(); });
822   $clearButton.bind('mouseup touchend', function(evt) { $input.val(null).trigger('keyup'); });
823   $overlay.bind('mousedown touchstart', function(evt) { evt.stopPropagation(); evt.preventDefault(); });
824   $overlay.bind('mouseup touchend', function(evt) { $input.trigger('blur'); });
825   
826   $input.bind('keyup', function(evt) {
827     
828     // If 'ESC' key was pressed, cancel the search.
829     if (evt.keyCode === 27) {
830       $cancelButton.trigger('mouseup');
831       return;
832     }
833     
834     var searchString = $input.val();
835     var tableView = self._tableView;
836     
837     if (!searchString) {
838       $overlay.addClass('pp-active');
839       $clearButton.removeClass('pp-active');
840     } else {
841       $overlay.removeClass('pp-active');
842       $clearButton.addClass('pp-active');
843     }
844     
845     if (tableView.getDataSource().shouldReloadTableForSearchString(searchString)) tableView.reloadData();
846   });
847   
848   this.attachToTableView(tableView);
849 };
