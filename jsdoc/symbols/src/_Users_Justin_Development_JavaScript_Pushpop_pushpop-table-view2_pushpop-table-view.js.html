  6 /**
  7   Creates a new TableView.
  8   @param {HTMLUListElement} element The UL element to initialize as a new TableView.
  9   @constructor
 10 */
 11 Pushpop.TableView = function TableView(element) {
 12   if (!element) return;
 13   
 14   var $element = this.$element = $(element);
 15   var element = this.element = $element[0];
 16   
 17   var tableView = element.tableView;
 18   if (tableView) return tableView;
 19   
 20   element.tableView = this;
 21   
 22   var scrollViewElement = $element.parents('.sk-scroll-view')[0];
 23   if (!scrollViewElement) return;
 24   
 25   var self = this;
 26   
 27   var reusableCells = this._reusableCells = {};
 28   var visibleCells = this._visibleCells = [];
 29   var selectedRowIndexes = this._selectedRowIndexes = [];
 30   
 31   var scrollView = this.scrollView = scrollViewElement.scrollView;
 32   
 33   var containsSearchBar = $element.attr('data-contains-search-bar') || 'false';
 34   containsSearchBar = containsSearchBar !== 'false';
 35   
 36   var searchBar = null;
 37   if (containsSearchBar) searchBar = this._searchBar = new Pushpop.TableViewSearchBar(this);
 38   
 39   var numberOfBufferedCells = this._numberOfBufferedCells;
 40   var selectionTimeoutDuration = this._selectionTimeoutDuration;
 41   var lastOffset = -scrollView.y;
 42   var topMargin = window.parseInt($element.css('margin-top'), 10);
 43   
 44   // Render table view cells "virtually" when the view is scrolled.
 45   scrollView.$element.bind(SKScrollEventType.ScrollChange, function(evt) {
 46     var offset = -scrollView.y;
 47     if (offset < 0) return;
 48     
 49     var firstCellElement = $element.children('li:first-child')[0];
 50     var lastCellElement = $element.children('li:last-child')[0];
 51     if (!firstCellElement || !lastCellElement || firstCellElement === lastCellElement) return;
 52     
 53     var dataSource = self.getDataSource();
 54     var rowHeight = self.getRowHeight();
 55     var numberOfRows = self._numberOfRows;
 56     var visibleCellCount = self.getCalculatedNumberOfVisibleCells();
 57     var selectedRowIndex = self.getIndexForSelectedRow();
 58     
 59     var firstCell = firstCellElement.tableViewCell;
 60     var firstCellIndex = firstCell.getIndex();
 61     var lastCellIndex = firstCellIndex + visibleCellCount - 1;
 62     
 63     // Manually calculate offset instead of calling .offset().
 64     var margin = scrollView.getMargin();
 65     var firstCellOffset = margin.top - offset;
 66     var lastCellOffset = firstCellOffset + (visibleCellCount * rowHeight);
 67     var delta = offset - lastOffset;
 68     var visibleHeight = self.getVisibleHeight();
 69     
 70     lastOffset = offset;
 71     
 72     var newMarginTopDelta = 0;
 73     var newMarginBottomDelta = 0;
 74     
 75     // Handle scrolling when swiping up (scrolling towards the bottom).
 76     if (delta > 0 && lastCellIndex + 1 < numberOfRows && firstCellOffset < 0 - (rowHeight * numberOfBufferedCells)) {
 77       $element.children('li:nth-child(-n+' + numberOfBufferedCells + ')').each(function(index, element) {
 78         var newCellIndex = lastCellIndex + index + 1;
 79         if (newCellIndex >= numberOfRows) return;
 80         
 81         var cell = element.tableViewCell;
 82         cell.prepareForReuse();
 83         
 84         var newCell = dataSource.getCellForRowAtIndex(self, newCellIndex);
 85         if (self.isRowSelectedAtIndex(newCellIndex)) newCell.setSelected(true);
 86         $element.append(newCell.$element);
 87         
 88         newMarginTopDelta += rowHeight;
 89         newMarginBottomDelta -= rowHeight;
 90       });
 91       
 92       scrollView.setMargin({
 93         top: margin.top + newMarginTopDelta,
 94         bottom: margin.bottom + newMarginBottomDelta
 95       });
 96     }
 97     
 98     // Handle scrolling when swiping down (scrolling towards the top).
 99     else if (delta < 0 && firstCellIndex - 1 >= 0 && lastCellOffset > visibleHeight + (rowHeight * numberOfBufferedCells)) {
100       $element.children('li:nth-child(n+' + (visibleCellCount - numberOfBufferedCells + 1) + ')').each(function(index, element) {
101         var newCellIndex = firstCellIndex - index - 1;
102         if (newCellIndex < 0) return;
103         
104         var cell = element.tableViewCell;
105         cell.prepareForReuse();
106         
107         var newCell = dataSource.getCellForRowAtIndex(self, newCellIndex);
108         if (self.isRowSelectedAtIndex(newCellIndex)) newCell.setSelected(true);
109         $element.prepend(newCell.$element);
110         
111         newMarginTopDelta -= rowHeight;
112         newMarginBottomDelta += rowHeight;
113       });
114       
115       scrollView.setMargin({
116         top: margin.top + newMarginTopDelta,
117         bottom: margin.bottom + newMarginBottomDelta
118       });
119     }
120   });
121   
122   // Handle case when table view is scrolled to the top (e.g.: tapping top of navigation bar).
123   scrollView.$element.bind(SKScrollEventType.WillScrollToTop, function(evt) {    
124     var firstCellElement = $element.children('li:first-child')[0];
125     if (firstCellElement) firstCellElement.tableViewCell.setIndex(0);
126   }).bind(SKScrollEventType.DidScrollToTop, function(evt) { self.reloadData(); });
127   
128   // Handle mouse/touch events to allow the user to tap accessory buttons.
129   var isPendingAccessoryButtonTap = false;
130 
131   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchstart' in window) ? 'touchstart' : 'mousedown', function(evt) {
132     isPendingAccessoryButtonTap = true;
133   });
134   
135   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchmove' in window) ? 'touchmove' : 'mousemove', function(evt) {
136     if (!isPendingAccessoryButtonTap) return;
137     isPendingAccessoryButtonTap = false;
138   });
139   
140   $element.delegate('span.pp-table-view-cell-accessory', !!('ontouchend' in window) ? 'touchend' : 'mouseup', function(evt) {
141     if (!isPendingAccessoryButtonTap) return;
142     isPendingAccessoryButtonTap = false;
143     
144     var tableViewCell = $(this).parent()[0].tableViewCell;
145     if (!tableViewCell) return;
146     
147     $element.trigger($.Event(Pushpop.TableView.EventType.AccessoryButtonTappedForRowWithIndex, {
148       tableView: self,
149       index: tableViewCell.getIndex()
150     }));
151   });
152   
153   // Handle mouse/touch events to allow the user to tap editing accessory buttons.
154   var isPendingEditingAccessoryButtonTap = false;
155 
156   $element.delegate('span.pp-table-view-cell-editing-accessory', !!('ontouchstart' in window) ? 'touchstart' : 'mousedown', function(evt) {
157     isPendingEditingAccessoryButtonTap = true;
158   });
159   
160   $element.delegate('span.pp-table-view-cell-editing-accessory', !!('ontouchmove' in window) ? 'touchmove' : 'mousemove', function(evt) {
161     if (!isPendingEditingAccessoryButtonTap) return;
162     isPendingEditingAccessoryButtonTap = false;
163   });
164   
165   $element.delegate('span.pp-table-view-cell-editing-accessory', !!('ontouchend' in window) ? 'touchend' : 'mouseup', function(evt) {
166     if (!isPendingEditingAccessoryButtonTap) return;
167     isPendingEditingAccessoryButtonTap = false;
168     
169     var tableViewCell = $(this).parent()[0].tableViewCell;
170     if (!tableViewCell) return;
171     
172     $element.trigger($.Event(Pushpop.TableView.EventType.EditingAccessoryButtonTappedForRowWithIndex, {
173       tableView: self,
174       index: tableViewCell.getIndex()
175     }));
176   });
177   
178   // Handle mouse/touch events to allow the user to make row selections.
179   var isPendingSelection = false, selectionTimeout = null;
180 
181   $element.delegate('li', !!('ontouchstart' in window) ? 'touchstart' : 'mousedown', function(evt) {
182     
183     // Don't allow row to be selected if an accessory button is pending a tap.
184     if (isPendingAccessoryButtonTap || isPendingEditingAccessoryButtonTap) return;
185     
186     isPendingSelection = true;
187     
188     var tableViewCell = this.tableViewCell;
189     
190     selectionTimeout = window.setTimeout(function() {
191       if (!isPendingSelection) return;
192       isPendingSelection = false;
193       
194       self.selectRowAtIndex(tableViewCell.getIndex());
195     }, selectionTimeoutDuration);
196   });
197   
198   $element.bind(!!('ontouchmove' in window) ? 'touchmove' : 'mousemove', function(evt) {
199     if (!isPendingSelection) return;
200     isPendingSelection = false;
201     
202     window.clearTimeout(selectionTimeout);
203   });
204   
205   $element.delegate('li', !!('ontouchend' in window) ? 'touchend' : 'mouseup', function(evt) {
206     if (!isPendingSelection) return;
207     isPendingSelection = false;
208     
209     window.clearTimeout(selectionTimeout);
210     self.selectRowAtIndex(this.tableViewCell.getIndex());
211   });
212   
213   // Create a new data source from a data set URL.
214   var dataSetUrl = $element.attr('data-set-url');
215   if (dataSetUrl) {
216     $.getJSON(dataSetUrl, function(dataSet) {
217       $element.html(null);
218       self.setDataSource(new Pushpop.TableViewDataSource(dataSet));
219     });
220   }
221   
222   // Create a new data source from existing <li/> elements.
223   else {
224     (function(self, $element) {
225       var dataSet = [];
226       
227       var dashAlpha = /-([a-z]|[0-9])/ig;
228       var camelCase = function(string) { return string.replace(dashAlpha, function(all, letter) { return (letter + '').toUpperCase(); }); };
229       
230       $element.children('li').each(function(index, element) {
231         var data = { title: $(element).html() };
232         
233         var attributes = element.attributes;
234         var attribute, attributeName;
235         for (var i = 0, length = attributes.length; i < length; i++) {
236           attribute = attributes[i];
237           attributeName = attribute.name;
238           
239           if (attributeName.indexOf('data-') === 0) data[camelCase(attributeName.substring(5))] = attribute.value;
240         }
241         
242         if (!data['reuseIdentifier']) data.reuseIdentifier = 'pp-table-view-cell-default';
243       
244         dataSet.push(data);
245       });
246       
247       $element.html(null);
248       self.setDataSource(new Pushpop.TableViewDataSource(dataSet));
249     })(self, $element);
250   }
251 };

264 Pushpop.TableView.getReusableCellPrototypes = function() {
265   var reusableCellPrototypes = Pushpop.TableView._reusableCellPrototypes, items = [];
266   for (var reusableCellPrototype in reusableCellPrototypes) items.push(reusableCellPrototypes[reusableCellPrototype]);
267   return items;

272 Pushpop.TableView.registerReusableCellPrototype = function(cellPrototype) {
273   if (!cellPrototype) return;
274   
275   var reuseIdentifier = cellPrototype.getReuseIdentifier();
276   if (!reuseIdentifier) return;
277   
278   Pushpop.TableView._reusableCellPrototypes[reuseIdentifier] = cellPrototype;

415   /**
416     Selects the row at the specified index and triggers the DidSelectRowAtIndex event on
417     this and all parent table view elements.
418     @description NOTE: If the row contains a child data source, this method will automatically
419     push a dynamic table view using the child data source. The DidSelectRowAtIndex event contains
420     a flag |hasChildDataSource| to indicate whether or not a new dynamic table view was pushed
421     prior to the event.
422     @param {Number} index The index of a row in the data source to select.
423     @param {Boolean} [animated] A flag indicating if the selection should be animated
424     if the row is currently visible.
425   */
426   selectRowAtIndex: function(index, animated) {
427     var dataSource = this.getDataSource();
428     var shouldSelectRowAtIndex = dataSource.shouldSelectRowAtIndex(index);
429     if (!shouldSelectRowAtIndex) return;
430     
431     this.deselectAllRows();
432     
433     var $element = this.$element;
434     this._selectedRowIndexes.push(index);
435     
436     var tableViewCell, $cells = this.$element.children();
437     for (var i = 0, length = $cells.length; i < length; i++) {
438       tableViewCell = $cells[i].tableViewCell;
439       if (tableViewCell.getIndex() === index) {
440         tableViewCell.setSelected(true);
441         break;
442       }
443     }
444     
445     // If this row contains a child data source, automatically push a new dynamic table view with it.
446     if (dataSource.rowHasChildDataSourceAtIndex(index)) {
447       var childDataSource = dataSource.getChildDataSourceForRowAtIndex(index);
448       var viewStack = this.getViewStack();
449       var self = this;
450       
451       if (childDataSource && viewStack) {
452         viewStack.pushNewTableView(function(childTableView) {
453           if (self.getSearchBar()) childTableView.setSearchBar(new Pushpop.TableViewSearchBar(childTableView));
454           childTableView.setDataSource(childDataSource);
455           childTableView.setParentTableView(self);
456         });
457         
458         // Trigger the DidSelectRowAtIndex event on this and all parent table view elements.
459         this.triggerEventOnParentTableViews($.Event(Pushpop.TableView.EventType.DidSelectRowAtIndex, {
460           tableView: this,
461           index: index,
462           hasChildDataSource: true
463         }), true);
464         
465         return;
466       }
467     }
468     
469     // Trigger the DidSelectRowAtIndex event on this and all parent table view elements.
470     this.triggerEventOnParentTableViews($.Event(Pushpop.TableView.EventType.DidSelectRowAtIndex, {
471       tableView: this,
472       index: index,
473       hasChildDataSource: false
474     }), true);
475   },

477   /**
478     De-selects the row at the specified index and optionally animates the de-selection
479     if the row is currently visible.
480     @description NOTE: This method will not modify any other existing selections.
481     @param {Number} index The index of a row in the data source to de-select.
482     @param {Boolean} [animated] A flag indicating if the de-selection should be
483     animated if the row is currently visible.
484   */
485   deselectRowAtIndex: function(index, animated) {
486     var $element = this.$element;
487     var selectedRowIndexes = this._selectedRowIndexes;
488     for (var i = 0, length = selectedRowIndexes.length; i < length; i++) {
489       if (selectedRowIndexes[i] === index) {
490         selectedRowIndexes.splice(i, 1);
491         break;
492       }
493     }
494     
495     var tableViewCell, $selectedCells = $element.children('.pp-table-view-selected-state');
496     for (var i = 0, length = $cells.length; i < length; i++) {
497       tableViewCell = $cells[i].tableViewCell;
498       if (tableViewCell.getIndex() === index) {
499         tableViewCell.setSelected(false);
500         break;
501       }
502     }
503     
504     $element.trigger($.Event(Pushpop.TableView.EventType.DidDeselectRowAtIndex, {
505       tableView: this,
506       index: index
507     }));
508   },
509   
510   /**
511     De-selects all rows in the table.
512   */
513   deselectAllRows: function() {
514     var $element = this.$element;
515     var selectedRowIndexes = this._selectedRowIndexes;
516     for (var i = 0, length = selectedRowIndexes.length; i < length; i++) {
517       $element.trigger($.Event(Pushpop.TableView.EventType.DidDeselectRowAtIndex, {
518         tableView: this,
519         index: selectedRowIndexes[i]
520       }));
521     }
522     
523     selectedRowIndexes.length = 0;
524     
525     $element.children('.pp-table-view-selected-state').each(function(index, element) {
526       element.tableViewCell.setSelected(false);
527     });
528   },

530   /**
531     Resets the scroll position back to the top of the TableView.
532   */
533   resetScrollView: function() {
534     var numberOfRows = this._numberOfRows;
535     var visibleCellCount = Math.min(this.getCalculatedNumberOfVisibleCells(), numberOfRows);
536     var hiddenCellCount = numberOfRows - visibleCellCount;
537     
538     var scrollView = this.scrollView;
539     
540     // Scroll to the top of the table view without animating.
541     scrollView.setContentOffset({ x: 0, y: 0 }, false);
542     
543     // Set the scroll view margin.
544     scrollView.setMargin({
545       top: 0,
546       bottom: hiddenCellCount * this.getRowHeight()
547     });
548   },

550   /**
551     Reloads the data source for the TableView and resets the scroll position back
552     to the top of the TableView.
553     @description Typically this method is called if/when the data source changes or
554     if there is a change in the search string.
555   */
556   reloadData: function() {
557     var $element = this.$element;    
558     
559     var dataSource = this.getDataSource();
560     var visibleCells = this.getVisibleCells();
561     
562     var i, length, visibleCellsToReuse = [];
563     for (i = 0, length = visibleCells.length; i < length; i++) visibleCellsToReuse.push(visibleCells[i]);
564     for (i = 0, length = visibleCellsToReuse.length; i < length; i++) visibleCellsToReuse[i].prepareForReuse();
565     
566     var numberOfRows = this._numberOfRows = dataSource.getNumberOfRows();
567     var visibleCellCount = Math.min(this.getCalculatedNumberOfVisibleCells(), numberOfRows);
568     var hiddenCellCount = numberOfRows - visibleCellCount;
569     
570     for (i = 0; i < visibleCellCount; i++) {
571       var cell = dataSource.getCellForRowAtIndex(this, i);      
572       $element.append(cell.$element);
573     }
574     
575     this.resetScrollView();
576   },
