  6 /**
  7   Creates a new PickerTableView.
  8   @param {HTMLDivElement} element The <div/> element to initialize as a new ModalViewStack.
  9   @constructor
 10 */
 11 Pushpop.PickerTableView = function PickerTableView(element) {
 12   
 13   // Call the "super" constructor.
 14   Pushpop.TableView.prototype.constructor.apply(this, arguments);
 15   
 16   // Put the table view in editing mode.
 17   this.$element.addClass('pp-table-view-editing');
 18   
 19   // Replace the existing data source with a PickerTableViewDataSource.
 20   var dataSource = this.getDataSource();
 21   var dataSet = dataSource.getDataSet();
 22   var defaultReuseIdentifier = dataSource.getDefaultReuseIdentifier();
 23   
 24   dataSource = new Pushpop.PickerTableViewDataSource(dataSet, defaultReuseIdentifier);
 25   this.setDataSource(dataSource);
 26   
 27   var self = this;
 28   
 29   // Listen for the "picker cell" to be selected and push a new table view with the picker
 30   // cell's data source to present a list of items to select from.
 31   this.$bind(Pushpop.TableView.EventType.DidSelectRowAtIndex, function(evt) {
 32     var dataSource = self.getDataSource();
 33     var index = evt.index;
 34     
 35     var pickerCellDataSource = self.getPickerCellDataSource();
 36     if (!pickerCellDataSource) return;
 37     
 38     var isPickerCell = (index === dataSource.getPickerCellIndex());
 39     if (!isPickerCell) return;
 40     
 41     var viewStack = self.getViewStack();
 42     if (!viewStack) return;
 43     
 44     var view = self.getView();
 45     if (!view) return;
 46     
 47     viewStack.pushNewTableView(function(newTableView) {
 48       newTableView.setSearchBar(new Pushpop.TableViewSearchBar(newTableView));
 49       newTableView.setDataSource(pickerCellDataSource);
 50       
 51       newTableView.$bind(Pushpop.TableView.EventType.DidSelectRowAtIndex, function(evt) {
 52         if (evt.hasChildDataSource) return;
 53         
 54         var tableView = evt.tableView;
 55         var pickerCellDataSource = tableView.getDataSource();
 56         var item = pickerCellDataSource.getFilteredItemAtIndex(evt.index);
 57         dataSource.addItem(item);
 58         viewStack.pop(view);
 59       });
 60     });
 61   });
 62   
 63   // Listen for tap events on the editing accessory buttons to add/remove items.
 64   this.$bind(Pushpop.TableView.EventType.EditingAccessoryButtonTappedForRowWithIndex, function(evt) {
 65     var dataSource = self.getDataSource();
 66     var index = evt.index;
 67     
 68     var isPickerCell = (index === dataSource.getPickerCellIndex());
 69     if (isPickerCell) return self.selectRowAtIndex(index);
 70     
 71     dataSource.removeItem(dataSource.getItemAtIndex(index));
 72   });
 73 };

 94 /**
 95   Creates a new data source for a PickerTableView.
 96   @param {Array} [dataSet] An optional array of data to initialize a default data source.
 97   @param {Array} [dataSet.id] The unique identifier for a specific record.
 98   @param {Array} [dataSet.value] The (sometimes) hidden value for a specific record.
 99   @param {Array} [dataSet.title] The title to be displayed in a TableViewCell for a specific record.
100   @param {String} [defaultReuseIdentifier] The optional reuse identifier to be used for rows that do
101   not specify a specific reuse identifier.
102   @constructor
103 */
104 Pushpop.PickerTableViewDataSource = function PickerTableViewDataSource(dataSet, defaultReuseIdentifier, pickerCellTitle) {
105   
106   // Call the "super" constructor.
107   Pushpop.TableViewDataSource.prototype.constructor.apply(this, arguments);
108   
109   // Set the picker cell's title.
110   this.setPickerCellTitle(pickerCellTitle || this.getPickerCellTitle());
111 };

113 // Create the prototype for the Pushpop.PickerTableViewDataSource as a "sub-class" of Pushpop.TableViewDataSource.
114 Pushpop.PickerTableViewDataSource.prototype = new Pushpop.TableViewDataSource();
115 Pushpop.PickerTableViewDataSource.prototype.constructor = Pushpop.PickerTableViewDataSource;
116 
117 /**
118   REQUIRED: Returns the number of rows provided by this data source.
119   @description NOTE: This is the default implementation and should be overridden for data
120   sources that are not driven directly from an in-memory data set. Also, in order to account
121   for the picker cell, this method should always return the number of rows in the data source
122   plus one.
123   @type Number
124 */
125 Pushpop.PickerTableViewDataSource.prototype.getNumberOfRows = function() { return this.getNumberOfFilteredItems() + 1; },
126 
127 /**
128   REQUIRED: Returns a TableViewCell for the specified index.
129   @description NOTE: This is the default implementation and should be overridden for data
130   sources that are not driven directly from an in-memory data set. Also, in order to account
131   for the picker cell, this method should always return the picker cell for the last available
132   index.
133   @param {Pushpop.TableView} tableView The TableView the TableViewCell should be returned for.
134   @param {Number} index The index of the data to be used when populating the TableViewCell.
135   @type Pushpop.TableViewCell
136 */
137 Pushpop.PickerTableViewDataSource.prototype.getCellForRowAtIndex = function(tableView, index) {
138   var isPickerCell = (index === this.getPickerCellIndex());
139   var item, reuseIdentifier, cell;
140   
141   item = (isPickerCell) ? {
142     title: this.getPickerCellTitle(),
143     reuseIdentifier: this.getPickerCellReuseIdentifier()
144   } : this.getFilteredItemAtIndex(index);
145   
146   reuseIdentifier = item.reuseIdentifier || this.getDefaultReuseIdentifier();
147   cell = tableView.dequeueReusableCellWithIdentifier(reuseIdentifier);
148   
149   cell.setIndex(index);
150   cell.setAccessoryType(item.accessoryType);
151   cell.setEditingAccessoryType(isPickerCell ? Pushpop.TableViewCell.EditingAccessoryType.AddButton : Pushpop.TableViewCell.EditingAccessoryType.DeleteButton);
152   cell.setData(item);
153   
154   return cell;
155 }

173 Pushpop.PickerTableViewDataSource.prototype.getPickerCellIndex = function() { return this.getNumberOfRows() - 1; };
174 
175 Pushpop.PickerTableViewDataSource.prototype.addItem = function(item) {
176   var numberOfItems = this.getNumberOfItems();
177   var dataSet = this.getDataSet();
178   for (var i = 0; i < numberOfItems; i++) if (this.getItemAtIndex(i) === item) return;
179   
180   dataSet.push(item);
181   this.setDataSet(dataSet);
182 };
183 
184 Pushpop.PickerTableViewDataSource.prototype.removeItem = function(item) {
185   var numberOfItems = this.getNumberOfItems();
186   var dataSet = this.getDataSet();
187   for (var i = 0; i < numberOfItems; i++) if (this.getItemAtIndex(i) === item) {
188     dataSet.splice(i, 1);
189     this.setDataSet(dataSet);
190     return;
191   }
192 };
