  6 /**
  7   Creates a new PickerTableView.
  8   @param {HTMLDivElement} element The <div/> element to initialize as a new ModalViewStack.
  9   @constructor
 10 */
 11 Pushpop.PickerTableView = function PickerTableView(element) {
 12   
 13   // Call the "super" constructor.
 14   Pushpop.TableView.prototype.constructor.apply(this, arguments);
 15   
 16   var dataSource = this.getDataSource();
 17   var dataSet = dataSource.getDataSet();
 18   var defaultReuseIdentifier = dataSource.getDefaultReuseIdentifier();
 19   
 20   dataSource = new Pushpop.PickerTableViewDataSource(dataSet, defaultReuseIdentifier);
 21   this.setDataSource(dataSource);
 22   
 23   console.log(dataSource);
 24 };

 28 Pushpop.PickerTableView.prototype.constructor = Pushpop.PickerTableView;

 30 /**
 31   Creates a new data source for a PickerTableView.
 32   @param {Array} [dataSet] An optional array of data to initialize a default data source.
 33   @param {Array} [dataSet.id] The unique identifier for a specific record.
 34   @param {Array} [dataSet.value] The (sometimes) hidden value for a specific record.
 35   @param {Array} [dataSet.title] The title to be displayed in a TableViewCell for a specific record.
 36   @param {String} [defaultReuseIdentifier] The optional reuse identifier to be used for rows that do
 37   not specify a specific reuse identifier.
 38   @constructor
 39 */
 40 Pushpop.PickerTableViewDataSource = function PickerTableViewDataSource(dataSet, defaultReuseIdentifier, pickerCellTitle) {
 41   
 42   // Call the "super" constructor.
 43   Pushpop.TableViewDataSource.prototype.constructor.apply(this, arguments);
 44   
 45   // Set the picker cell's title.
 46   this.setPickerCellTitle(pickerCellTitle || this.getPickerCellTitle());
 47 };

 49 // Create the prototype for the Pushpop.PickerTableViewDataSource as a "sub-class" of Pushpop.TableViewDataSource.
 50 Pushpop.PickerTableViewDataSource.prototype = new Pushpop.TableViewDataSource();
 51 Pushpop.PickerTableViewDataSource.prototype.constructor = Pushpop.PickerTableViewDataSource;
 52 
 53 /**
 54   REQUIRED: Returns the number of rows provided by this data source.
 55   @description NOTE: This is the default implementation and should be overridden for data
 56   sources that are not driven directly from an in-memory data set. Also, in order to account
 57   for the picker cell, this method should always return the number of rows in the data source
 58   plus one.
 59   @type Number
 60 */
 61 Pushpop.PickerTableViewDataSource.prototype.getNumberOfRows = function() { return this.getFilteredDataSet().length + 1; },
 62 
 63 /**
 64   REQUIRED: Returns a TableViewCell for the specified index.
 65   @description NOTE: This is the default implementation and should be overridden for data
 66   sources that are not driven directly from an in-memory data set. Also, in order to account
 67   for the picker cell, this method should always return the picker cell for the last available
 68   index.
 69   @param {Pushpop.TableView} tableView The TableView the TableViewCell should be returned for.
 70   @param {Number} index The index of the data to be used when populating the TableViewCell.
 71   @type Pushpop.TableViewCell
 72 */
 73 Pushpop.PickerTableViewDataSource.prototype.getCellForRowAtIndex = function(tableView, index) {
 74   var numberOfRows = this.getNumberOfRows();
 75   var data, reuseIdentifier, cell;
 76   
 77   data = (index === numberOfRows - 1) ? {
 78     title: this.getPickerCellTitle(),
 79     reuseIdentifier: this.getPickerCellReuseIdentifier()
 80   } : this.getFilteredDataSet()[index];
 81   
 82   reuseIdentifier = data.reuseIdentifier || this.getDefaultReuseIdentifier();
 83   cell = tableView.dequeueReusableCellWithIdentifier(reuseIdentifier);
 84   
 85   cell.setIndex(index);
 86   cell.setAccessoryType(data.accessoryType);
 87   cell.setData(data);
 88   
 89   return cell;
 90 }
