  6 /**
  7   Creates a new ModalViewStack.
  8   @param {HTMLDivElement} element The <div/> element to initialize as a new ModalViewStack.
  9   @constructor
 10 */
 11 Pushpop.ModalViewStack = function ModalViewStack(element) {
 12   
 13   // Call the "super" constructor.
 14   Pushpop.ViewStack.prototype.constructor.apply(this, arguments);
 15   
 16   var $element = this.$element;
 17   
 18   // Insert the backdrop element after the modal view stack element.
 19   var $backdrop = $('<div class="pp-modal-view-stack-backdrop"/>').insertAfter($element);
 20   
 21   // Initialize the modal view stack's presentation style based on the element's CSS classes.
 22   var presentationStyles = Pushpop.ModalViewStack.PresentationStyleType, didInitPresentationStyle = false;
 23   for (var presentationStyle in presentationStyles) {
 24     if ($element.hasClass('pp-modal-presentation-style-' + presentationStyles[presentationStyle])) {
 25       this.setPresentationStyle(presentationStyles[presentationStyle]);
 26       didInitPresentationStyle = true;
 27       break;
 28     }
 29   }
 30   
 31   // Set the modal view stack's presentation style to the default if one was not specified.
 32   if (!didInitPresentationStyle) this.setPresentationStyle(this.constructor.prototype._presentationStyle);
 33   
 34   // Initialize the modal view stack's transition style based on the element's CSS classes.
 35   var transitionStyles = Pushpop.ModalViewStack.TransitionStyleType, didInitTransitionStyle = false;
 36   for (var transitionStyle in transitionStyles) {
 37     if ($element.hasClass('pp-modal-transition-style-' + transitionStyles[transitionStyle])) {
 38       this.setTransitionStyle(transitionStyles[transitionStyle]);
 39       didInitTransitionStyle = true;
 40       break;
 41     }
 42   }
 43   
 44   // Set the modal view stack's transition style to the default if one was not specified.
 45   if (!didInitTransitionStyle) this.setTransitionStyle(this.constructor.prototype._transitionStyle);
 46   
 47   // Make the modal view stack visible if it is initialized with the .pp-active CSS class.
 48   this.setHidden(!$element.hasClass('pp-active'));
 49 };

 51 /**
 52   Presentation styles available when presenting a Pushpop.ModalViewStack.
 53 */
 54 Pushpop.ModalViewStack.PresentationStyleType = {
 55   FullScreen: 'full-screen',  // 100% x 100%
 56   PageSheet: 'page-sheet',    // 768px x 100%
 57   FormSheet: 'form-sheet'     // 540px x 620px
 58 };

 61   Transition styles available when presenting a Pushpop.ModalViewStack.

 64   CoverVertical: 'cover-vertical'

 67 // Create the prototype for the Pushpop.ModalViewStack as a "sub-class" of Pushpop.ViewStack.
 68 Pushpop.ModalViewStack.prototype = new Pushpop.ViewStack();
 69 Pushpop.ModalViewStack.prototype.constructor = Pushpop.ModalViewStack;

 71 /**
 72   Pushes the specified view to the view stack using the optionally specified transition. If a
 73   transition is not specified, the default will be used. A callback may optionally be provided
 74   to be called after the transition completes. In the case of a Pushpop.ModalViewStack, the
 75   modal view stack will first be presented if it is currently hidden.
 76   @description This method overrides the default implementation of Pushpop.ViewStack to present
 77   the modal view stack if it is hidden. After presenting the modal view stack (if needed), the
 78   "super" implementation of this method is then executed.
 79   @param {Pushpop.View} view The view to be pushed to the view stack.
 80   @param {String|Function} [transitionOrCallback] Either the name of the transition to use when
 81   pushing the view or a callback function to be executed upon completion of the default transition.
 82   If this parameter is omitted, the default transition is used.
 83   @param {Function} [callback] A callback function to be executed upon completion of the specified
 84   transition.
 85 */
 86 Pushpop.ModalViewStack.prototype.push = function(view, transitionOrCallback, callback) {
 87   if (this.getHidden()) this.present();
 88   
 89   // Call the "super" method.
 90   Pushpop.ViewStack.prototype.push.apply(this, arguments);
 91 };
 92 
 93 /**
 94   Pops the current or specified view off the view stack using the optionally specified transition.
 95   If a view is not specified, the current view will be popped (unless it is the root view). If a 
 96   transition is not specified, the default will be used. A callback may optionally be provided to
 97   be called after the transition completes. In the case of a Pushpop.ModalViewStack, the modal
 98   view stack will be dismissed if this method is called when there is only one remaining view on
 99   the view stack (e.g.: the "root" view) and no view is specified.
100   @description This method overrides the default implementation of Pushpop.ViewStack to dismiss
101   the modal view stack if there is only one remaining view on the view stack and no view is
102   specified. After dismissing the modal view stack (if needed), the "super" implementation of this
103   method is then executed.
104   @param {Pushpop.View|String} viewOrTransition Either the view to be popped to on the view stack
105   or the name of the transition to use when popping the view. If this parameter is omitted or if
106   it specifies a transition name, the current view will be assumed to be popped.
107   @param {String|Function} [transitionOrCallback] Either the name of the transition to use when
108   popping the view or a callback function to be executed upon completion of the default transition.
109   If this parameter is omitted, the default transition is used.
110   @param {Function} [callback] A callback function to be executed upon completion of the specified
111   transition.
112 */
113 Pushpop.ModalViewStack.prototype.pop = function(viewOrTransition, transitionOrCallback, callback) {
114   if (this.views.length <= 1) this.dismiss();
115   
116   // Call the "super" method.
117   Pushpop.ViewStack.prototype.pop.apply(this, arguments);
118 };

120 Pushpop.ModalViewStack.prototype._hidden = true;
121 
122 /**
123   Returns a flag indicating if this modal view stack is hidden.
124   @type Boolean
125 */
126 Pushpop.ModalViewStack.prototype.getHidden = function() { return this._hidden; };
127 
128 /**
129   Sets a flag indicating if this modal view stack should be hidden.
130   @param {Boolean} hidden A flag indicating if this modal view stack should be hidden.
131 */
132 Pushpop.ModalViewStack.prototype.setHidden = function(hidden) { if (hidden) this.dismiss(); else this.present(); };
133 
134 /**
135   Presents this modal view stack and transitions it to a visible state.
136 */
137 Pushpop.ModalViewStack.prototype.present = function() { this._hidden = false; this.$element.addClass('pp-active'); };
138 
139 /**
140   Dismisses this modal view stack and transitions it to a hidden state.
141 */
142 Pushpop.ModalViewStack.prototype.dismiss = function() { this._hidden = true; this.$element.removeClass('pp-active'); };
143 
144 Pushpop.ModalViewStack.prototype._presentationStyle = Pushpop.ModalViewStack.PresentationStyleType.FormSheet;
145 
146 /**
147   Returns the presentation style of this modal view stack.
148   @type String
149 */
150 Pushpop.ModalViewStack.prototype.getPresentationStyle = function() { return this._presentationStyle; };
151 
152 /**
153   Sets the presentation style to use when presenting this modal view stack.
154   @param {String} presentationStyle The presentation style for this modal view stack.
155 */
156 Pushpop.ModalViewStack.prototype.setPresentationStyle = function(presentationStyle) {
157   this._presentationStyle = presentationStyle;
158   
159   var $element = this.$element, presentationStyles = Pushpop.ModalViewStack.PresentationStyleType;
160   for (var p in presentationStyles) $element.removeClass('pp-modal-presentation-style-' + presentationStyles[p]);
161   $element.addClass('pp-modal-presentation-style-' + presentationStyle);
162 };

172 /**
173   Sets the transition style to use when presenting this modal view stack.
174   @param {String} transitionStyle The transition style for this modal view stack.
175 */
176 Pushpop.ModalViewStack.prototype.setTransitionStyle = function(transitionStyle) {
177   this._transitionStyle = transitionStyle;
178   
179   var $element = this.$element, transitionStyles = Pushpop.ModalViewStack.TransitionStyleType;
180   for (var t in transitionStyles) $element.removeClass('pp-modal-transition-style-' + transitionStyles[t]);
181   $element.addClass('pp-modal-transition-style-' + transitionStyle);
182 };

184 $(function() {
185   $('.pp-modal-view-stack').each(function(index, element) { new Pushpop.ModalViewStack(element); });
186   
187   $('a.pp-present-modal').live('click', function(evt) {
188     evt.preventDefault();
189     
190     var $this = $(this);
191     var href = $this.attr('href');
192     var $viewElement = $(href);
193     
194     var view = $viewElement[0];
195     if (view) view = view.view || new Pushpop.View($viewElement);
196     
197     var viewStack = view.getViewStack();    
198     if (viewStack) viewStack.present();
199   });
200   
201   $('a.pp-dismiss-modal').live('click', function(evt) {
202     evt.preventDefault();
203     
204     var $this = $(this);
205     var href = $this.attr('href');
206     var $viewElement, view, viewStack;
207     
208     if (href === '#') {
209       $viewElement = $this.parents('.pp-view').first();
210     } else {
211       $viewElement = $(href);
212     }
213     
214     if ($viewElement.length > 0) view = $viewElement[0].view;
215     
216     view = view || new Pushpop.View($viewElement);
217     
218     viewStack = view.getViewStack();      
219     if (viewStack) viewStack.dismiss();
220   });
221 });
